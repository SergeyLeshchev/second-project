pwd	показать где сейчас находишься(директория)
cd	сменить директорию
ls	вывести содержимое текущей директории
ls -a	вывести содержимое со скрытыми файлами и папками, а так же . и .. обозначающие текущую и родительскую директории
touch (имя файла.расширение)	создать новый файл

mkdir	создать директорию
mkdir -p	создать очередь директорий, одну в другой
cp что_копируем куда_копируем	копирование
mv	переместить
cat file	прочитать текстовый файл
rm example.txt	удалили файл example.txt из текущей папки
rmdir images    	команда удалит папку images из текущей директории, если папка images пуста
rm -r images    	удалили папку images со всем её содержимым из текущей директории
	
Выполняйте сразу несколько команд
mkdir second-project && cd second-project && touch index.html style.css
Вызывайте команды из буфера
Нажмите клавишу вверх, вниз.
Нажмите Tab для автозаполнения.
cd /c – перейти на диск С
cd – перейти в корневую директорию
git config --global user.name "User Namovich" – указать имя для работы в гите с командой
git config --global user.email username@yandex.ru – указать эл. Почту для работы в команде
cat ~/.gitconfig – проверить глобальные настройки
git config –list – то же самое
git init – чтобы создать репозиторий, в папке с файлами проекта нужно ввести команду.
rm -rf .git – внутри папки репозитория удаляется скрытый файл .git и папка перестает быть репозиторием(разгичивается).
git add --all – добавить к отслеживанию все файлы в этой папке
git add . – добавить к отслеживанию текущую папку и все файлы
git add file.txt – добавить к отслеживанию один файл
git commit -m ‘Message’ – сделать коммит с сообщением
git log – просмотреть историю коммитов
ls -la .ssh/ или ls -a .ssh/ - проверить наличие ssh-ключей
ssh-keygen -t ed25519 -C "email" или 
ssh-keygen -t rsa -b 4096 -C "email" – сгенерировать ssh-ключи. Потом Enter, enter и снова проверить наличие ключей.
clip < ~/.ssh/id_ed25519.pub – скопировать содержимое файла ключа
если не сработает, то:
cat ~/.ssh/id_rsa.pub
или cat ~/.ssh/id_ed25519.pub
git remote add origin ссылка с гитхаба
git remote -v – проверка подключения к удаленному репозиторию.\
